//= Shadow Item Trader
//===== By: rAthena AI Assistant ============================
//= Provides a shop for Shadow Items.

    switch(select("Costume Shadow Item")) {
        case 1:
            callshop "cShadowItem",0;
            end;
            break;
    }
    OnInit:
        waitingroom "Shadow Item Trader",0;
    end;
}
